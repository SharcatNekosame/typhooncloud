import requests import folium from folium.plugins import TimestampedGeoJson # 1. 設定 API 網址 url = "https://opendata.cwa.gov.tw/api/v1/rest/datastore/W-C0034-001?Authorization=CWA-BBE46FCC-A1BF-4EAC-A234-A68A5234042C&format=JSON" try: response = requests.get(url) data = response.json() # 取得颱風消息 typhoons = data['records']['tropicalCyclones']['tropicalCyclone'] if not typhoons: print("目前可能沒有發布中的颱風資料。") else: for info in typhoons: typhoon_name = info['typhoonName'] print(f"正在處理颱風：{typhoon_name}") # 建立地圖底圖（初始中心設在台灣附近） m = folium.Map(location=[23.5, 121.0], zoom_start=5, control_scale=True) # 用來儲存路徑軌跡的經緯度點 path_points = [] # 解析觀測（過去）與預測（未來）資料 analysis_data = info.get('analysis_and_forecast', {}).get('analysis', []) forecast_data = info.get('analysis_and_forecast', {}).get('forecast', []) # 合併所有點位進行繪製 all_points = analysis_data + forecast_data for point in all_points: lat = float(point['position']['latitude']) lng = float(point['position']['longitude']) time_str = point['time']['obsTime'] if 'obsTime' in point['time'] else point['time'].get('forecastTime', '') pressure = point.get('pressure', 'N/A') max_wind = point.get('maxWindSpeed', 'N/A') path_points.append([lat, lng]) # 取得七級風暴風半徑 (轉換為公尺) radius_7_km = point.get('circleOf7ms', {}).get('radius', 0) radius_7_m = float(radius_7_km) * 1000 if radius_7_km else 0 # 在地圖上畫出暴風圈 (七級風半徑) if radius_7_m > 0: folium.Circle( location=[lat, lng], radius=radius_7_m, color='red', fill=True, fill_color='red', fill_opacity=0.1, popup=f"暴風半徑: {radius_7_km} km" ).add_to(m) # 標記颱風中心點 popup_text = f""" 颱風名稱: {typhoon_name}
時間: {time_str}
中心氣壓: {pressure} hPa
近中心最大風速: {max_wind} m/s
位置: 北緯 {lat}, 東經 {lng} """ folium.Marker( location=[lat, lng], popup=folium.Popup(popup_text, max_width=300), icon=folium.Icon(color='blue' if 'obsTime' in point['time'] else 'green', icon='info-sign') ).add_to(m) # 畫出颱風移動路徑連線 if len(path_points) > 1: folium.PolyLine(path_points, color='black', weight=2.5, opacity=0.8).add_to(m) # 儲存為 HTML 網頁檔案 output_filename = f"typhoon_{typhoon_name}_map.html" m.save(output_filename) print(f"✨ 可視化地圖已成功生成！請打開同目錄下的檔案：{output_filename}") except Exception as e: print(f"讀取或解析資料時發生錯誤：{e}")
